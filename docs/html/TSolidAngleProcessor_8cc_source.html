94 void TSolidAngleProcessor::Process() {
95  if ((*fInData_cm)->GetEntriesFast() != 1) {
96  SetStateError("input reaction data branch entry is not 1");
97  return;
98  }
99 
100  const TDataObject *const inData = static_cast<TDataObject *>((*fInData_cm)->At(0));
101  const TReactionInfo *const Data = dynamic_cast<const TReactionInfo *>(inData);
102  h1_e_all->Fill(Data->GetEnergy());
103  for (Int_t i = 0; i < fNbin_angle; i++) {
104  Double_t angle_center = fRange_angle[0] + (fRange_angle[1] - fRange_angle[0]) / (2.0 * fNbin_angle) +
105  (Double_t)i * (fRange_angle[1] - fRange_angle[0]) / fNbin_angle;
106  h2_all->Fill(Data->GetEnergy(), angle_center);
107  }
108 
109  if ((*fInData_det)->GetEntriesFast() > 0) {
110  if ((*fInData_det)->GetEntriesFast() > 1) {
111  std::cerr << "not single particle detected" << std::endl;
112  return;
113  }
114 
115  h1_e->Fill(Data->GetEnergy());
116  if (fIsIK) {
117  h2->Fill(Data->GetEnergy(), 180.0 - Data->GetTheta()); // assume inverse kinematics
118  } else {
119  h2->Fill(Data->GetEnergy(), Data->GetTheta());
120  }
121  }
