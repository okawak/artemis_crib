218 Double_t TTGTIKProcessor::bisection(const TTrack *track, const TTelescopeData *data) {
219  Double_t flow = TargetFunction(kInitialMin, track, data);
220  Double_t fhigh = TargetFunction(kInitialMax, track, data);
221  Double_t low = kInitialMin;
222  Double_t high = kInitialMax;
223  Double_t ftmp = TargetFunction(kInitialMin, track, data);
224  Int_t sign = 1;
225  Bool_t isFirst = true;
226 
227  // search initial z position (rough bisection)
228  for (Int_t i = 0; i < 10; i++) {
229  Double_t tmpz = kInitialMin + (Double_t)i * (kInitialMax - kInitialMin);
230  Double_t f = TargetFunction(tmpz, track, data);
231  if (!IsValid(f))
232  continue;
233 
234  if (isFirst) {
235  sign = f >= 0 ? 1 : -1;
236  isFirst = false;
237  }
238 
239  if (sign == (f >= 0 ? 1 : -1)) {
240  ftmp = f;
241  } else {
242  flow = ftmp;
243  fhigh = f;
244  low = kInitialMin + (Double_t)(i - 1) * (kInitialMax - kInitialMin);
245  high = tmpz;
246  break;
247  }
248  }
249  if (!IsValid(flow) || !IsValid(fhigh)) {
250  std::cerr << "[Warning in bisect] f(low) or f(high) is invalid" << std::endl;
251  return kInvalidD;
252  }
253  if (flow * fhigh > 0) {
254  std::cerr << "[Warning in bisect] f(low) * f(high) > 0, f(low): " << flow << ", f(high): " << fhigh << std::endl;
255  return kInvalidD;
256  }
257 
258  // start standard bisection method
259  Double_t left = low;
260  Double_t right = high;
261 
262  Double_t middle = 0.0;
263  Int_t itr = 0;
264  while (TMath::Abs(right - left) > kEpsilon) {
265  middle = (left + right) / 2.0;
266 
267  Double_t fleft = TargetFunction(left, track, data);
268  Double_t fmiddle = TargetFunction(middle, track, data);
269  if (!IsValid(fleft) || !IsValid(fmiddle))
270  return kInvalidD;
271 
272  if (fleft * fmiddle < 0) {
273  right = middle;
274  } else {
275  left = middle;
276  }
277 
278  itr++;
279  if (itr == kMaxIteration) {
280  std::cerr << "[bisection] not converge!" << std::endl;
281  return kInvalidD;
282  }
283  }
284  return middle;
285 }

313 Double_t TTGTIKProcessor::GetEcmFromBeam(Double_t z, const TTrack *track) {
314  // if sign > 0 enewz, sign < 0, eoldz (thickness is negative)
315  Int_t sign = z > 0 : 1 ? -1;
316  TVector3 beam_flight(track->GetX(z) - track->GetX(0), track->GetY(z) - track->GetY(0), z);
317  Double_t energy = srim->EnergyNew(fParticleZArray[0], fParticleAArray[0],
318  fInitialBeamEnergy, std::string(fTargetName.Data()),
319  sign * beam_flight.Mag());
320  if (energy < 0.01)
321  return 0.0;
322 
323  Double_t beta = TMath::Sqrt(1.0 - TMath::Power(fParticleMass[0] / (fParticleMass[0] + energy), 2));
324  Double_t ang[2] = {track->GetA(), track->GetB()};
325  Double_t norm = TMath::Sqrt(TMath::Tan(ang[0]) * TMath::Tan(ang[0]) + TMath::Tan(ang[1]) * TMath::Tan(ang[1]) + 1.0);
326  Double_t beta_array[3] = {beta * TMath::Tan(ang[0]) / norm, beta * TMath::Tan(ang[1]) / norm, beta * 1.0 / norm};
327 
328  TLorentzVector beam(0., 0., 0., fParticleMass[0]);
329  beam.Boost(beta_array[0], beta_array[1], beta_array[2]);
330 
331  TLorentzVector target(0., 0., 0., fParticleMass[1]);
332 
333  TLorentzVector cm_vec = beam + target;
334  TVector3 beta_cm = cm_vec.BoostVector();
335  beam.Boost(-beta_cm);
336  target.Boost(-beta_cm);
337 
338  return (beam.E() - beam.M()) + (target.E() - target.M());
339 }

426 Double_t TTGTIKProcessor::GetEcm_classic_kinematics(Double_t energy, Double_t theta) {
427  // unit = mass (MeV)
428  Double_t alpha = (fParticleMass[1] * (fParticleMass[0] + fParticleMass[1])) / (2.0 * fParticleMass[0]);
429  Double_t beta = (fParticleMass[3] * (fParticleMass[2] + fParticleMass[3])) / (2.0 * fParticleMass[2]);
430  Double_t qvalue = (fParticleMass[0] + fParticleMass[1]) - (fParticleMass[2] + fParticleMass[3]);
431 
432  Double_t v4 = TMath::Sqrt(2.0 * energy / fParticleMass[3]);
433  Double_t b = (beta * v4 * TMath::Cos(theta)) / (alpha - beta);
434  Double_t c = (qvalue - beta * v4 * v4) / (alpha - beta);
435  Double_t D = b * b - c;
436  if (D < 0) {
437  if (TMath::Abs(D) < 1.0e-5) {
438  D = 0.0;
439  } else {
440  std::cout << "b^2 - c = " << D << " < 0, det_energy : " << energy << ", theta : " << theta << std::endl;
441  return kInvalidD;
442  }
443  }
444 
445  Double_t vcm = -b + TMath::Sqrt(D);
446  if (vcm < 0) {
447  std::cout << "vcm < 0! : vcm = " << -b << " + " << TMath::Sqrt(D) << ", det_energy : " << energy
448  << ", theta : " << theta << std::endl;
449  return kInvalidD;
450  }
451 
452  return alpha * vcm * vcm;
453 }
