170 Double_t TTGTIKProcessor::bisection(const TTrack *track, const TTelescopeData *data) {
171  Double_t flow = TargetFunction(-kWindowUncertainty, track, data);
172  Double_t fhigh = TargetFunction(fMaxLength, track, data);
173  Double_t low = -kWindowUncertainty;
174  Double_t high = fMaxLength;
175  Double_t ftmp = TargetFunction(-kWindowUncertainty, track, data);
176  Int_t sign = 1;
177  Bool_t isFirst = true;
178 
179  for (Int_t i = 0; i < 10; i++) {
180  Double_t tmpz = -kWindowUncertainty + (Double_t)i * (fMaxLength + kWindowUncertainty);
181  Double_t f = TargetFunction(tmpz, track, data);
182  if (!IsValid(f))
183  continue;
184 
185  if (isFirst) {
186  sign = f >= 0 ? 1 : -1;
187  isFirst = false;
188  }
189 
190  if (sign == f >= 0 ? 1 : -1) {
191  ftmp = f;
192  } else {
193  flow = ftmp;
194  fhigh = f;
195  low = -kWindowUncertainty + (Double_t)(i - 1) * (fMaxLength + kWindowUncertainty);
196  high = tmpz;
197  break;
198  }
199  }
200  if (!IsValid(flow) || !IsValid(fhigh))
201  return kInvalidD;
202  if (flow * fhigh > 0) {
203  std::cerr << "[Warning in bisect] f(low) * f(high) > 0, f(low): " << flow << ", f(high): " << fhigh << std::endl;
204  return kInvalidD;
205  }
206 
207  Double_t left = low;
208  Double_t right = high;
209 
210  Double_t middle = 0.0;
211  Int_t itr = 0;
212  while (TMath::Abs(right - left) > kEpsilon) {
213  middle = (left + right) / 2.0;
214 
215  Double_t fleft = TargetFunction(left, track, data);
216  Double_t fmiddle = TargetFunction(middle, track, data);
217  if (!IsValid(fleft) || !IsValid(fmiddle))
218  return kInvalidD;
219 
220  if (fleft * fmiddle < 0) {
221  right = middle;
222  } else {
223  left = middle;
224  }
225 
226  itr++;
227  if (itr == kMaxIteration) {
228  std::cerr << "[bisection] not converge!" << std::endl;
229  return kInvalidD;
230  }
231  }
232  return middle;
233 }

250 Double_t TTGTIKProcessor::GetEcmFromBeam(Double_t z, const TTrack *track) {
251  Double_t energy = 0.0;
252  TVector3 beam_flight(track->GetX(z) - track->GetX(0), track->GetY(z) - track->GetY(0), z);
253  if (z > 0) {
254  energy = fElossBeam->GetNewE(fInitialBeamEnergy, beam_flight.Mag());
255  if (energy < 0.01)
256  return 0.0;
257  } else if (z < 0) {
258  energy = fElossBeam->GetOldE(fInitialBeamEnergy, beam_flight.Mag());
259  }
260 
261  Double_t beta = TMath::Sqrt(1.0 - TMath::Power(fParticleMass[0] / (fParticleMass[0] + energy), 2));
262  Double_t ang[2] = {track->GetA(), track->GetB()};
263  Double_t norm = TMath::Sqrt(TMath::Tan(ang[0]) * TMath::Tan(ang[0]) + TMath::Tan(ang[1]) * TMath::Tan(ang[1]) + 1.0);
264  Double_t beta_array[3] = {beta * TMath::Tan(ang[0]) / norm, beta * TMath::Tan(ang[1]) / norm, beta * 1.0 / norm};
265 
266  TLorentzVector beam(0., 0., 0., fParticleMass[0]);
267  beam.Boost(beta_array[0], beta_array[1], beta_array[2]);
268 
269  TLorentzVector target(0., 0., 0., fParticleMass[1]);
270 
271  TLorentzVector cm_vec = beam + target;
272  TVector3 beta_cm = cm_vec.BoostVector();
273  beam.Boost(-beta_cm);
274  target.Boost(-beta_cm);
275 
276  return (beam.E() - beam.M()) + (target.E() - target.M());
277 }

346 Double_t TTGTIKProcessor::GetEcm_classic_kinematics(Double_t energy, Double_t theta) {
347  // unit = mass (MeV)
348  Double_t alpha = (fParticleMass[1] * (fParticleMass[0] + fParticleMass[1])) / (2.0 * fParticleMass[0]);
349  Double_t beta = (fParticleMass[3] * (fParticleMass[2] + fParticleMass[3])) / (2.0 * fParticleMass[2]);
350  Double_t qvalue = (fParticleMass[0] + fParticleMass[1]) - (fParticleMass[2] + fParticleMass[3]);
351 
352  Double_t v4 = TMath::Sqrt(2.0 * energy / fParticleMass[3]);
353  Double_t b = (beta * v4 * TMath::Cos(theta)) / (alpha - beta);
354  Double_t c = (qvalue - beta * v4 * v4) / (alpha - beta);
355  Double_t D = b * b - c;
356  if (D < 0) {
357  if (TMath::Abs(D) < 1.0e-5) {
358  D = 0.0;
359  } else {
360  std::cout << "b^2 - c = " << D << " < 0, det_energy : " << energy << ", theta : " << theta << std::endl;
361  return kInvalidD;
362  }
363  }
364 
365  Double_t vcm = -b + TMath::Sqrt(D);
366  if (vcm < 0) {
367  std::cout << "vcm < 0! : vcm = " << -b << " + " << TMath::Sqrt(D) << ", det_energy : " << energy
368  << ", theta : " << theta << std::endl;
369  return kInvalidD;
370  }
371 
372  return alpha * vcm * vcm;
373 }
