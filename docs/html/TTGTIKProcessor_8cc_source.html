254 Double_t TTGTIKProcessor::bisection(const TTrack *track, const TTelescopeData *data) {
255  Double_t flow = TargetFunction(kInitialMin, track, data);
256  Double_t fhigh = TargetFunction(kInitialMax, track, data);
257  Double_t low = kInitialMin;
258  Double_t high = kInitialMax;
259  Double_t ftmp = TargetFunction(kInitialMin, track, data);
260 
261  // search initial z position (rough bisection)
262  Int_t step = 10;
263  for (Int_t i = 0; i < step + 1; i++) {
264  Double_t tmpz = kInitialMin + (Double_t)i * (kInitialMax - kInitialMin) / (Double_t)step;
265  Double_t f = TargetFunction(tmpz, track, data);
266  if (!IsValid(f))
267  continue;
268 
269  if (ftmp * f > 0.0) {
270  ftmp = f;
271  } else {
272  flow = ftmp;
273  fhigh = f;
274  low = kInitialMin + (Double_t)(i - 1) * (kInitialMax - kInitialMin) / (Double_t)step;
275  high = tmpz;
276  break;
277  }
278  }
279  if (!IsValid(flow) || !IsValid(fhigh)) {
280  std::cerr << "[Warning in bisect] f(low) or f(high) is invalid" << std::endl;
281  return kInvalidD;
282  }
283  if (flow * fhigh > 0) {
284  std::cerr << "[Warning in bisect] f(low) * f(high) > 0"
285  << ", f(low): " << flow << ", f(high): " << fhigh
286  << ", energy = " << data->GetEtotal()
287  << std::endl;
288  return kInvalidD;
289  }
290  if (low == kInitialMin || high == kInitialMax) {
291  std::cerr << "[Warning in bisect] could not find =0 position" << std::endl;
292  return kInvalidD;
293  }
294 
295  // start standard bisection method
296  Double_t left = low;
297  Double_t right = high;
298 
299  Double_t middle = 0.0;
300  Int_t itr = 0;
301  while (TMath::Abs(right - left) > kEpsilon) {
302  middle = (left + right) / 2.0;
303 
304  Double_t fleft = TargetFunction(left, track, data);
305  Double_t fmiddle = TargetFunction(middle, track, data);
306  if (!IsValid(fleft) || !IsValid(fmiddle))
307  return kInvalidD;
308 
309  if (fleft * fmiddle < 0) {
310  right = middle;
311  } else {
312  left = middle;
313  }
314 
315  itr++;
316  if (itr == kMaxIteration) {
317  std::cerr << "[bisection] not converge!" << std::endl;
318  return kInvalidD;
319  }
320  }
321  return middle;
322 }

347 Double_t TTGTIKProcessor::GetEcmFromBeam(Double_t z, const TTrack *track) {
348  // if sign > 0 enewz, sign < 0, eoldz (thickness is negative)
349  Int_t sign = z > 0 ? 1 : -1;
350  TVector3 beam_flight(track->GetX(z) - track->GetX(0), track->GetY(z) - track->GetY(0), z);
351  Double_t energy = srim->EnergyNew(fParticleZArray[0], fParticleAArray[0],
352  fInitialBeamEnergy, std::string(fTargetName.Data()),
353  sign * beam_flight.Mag(), fPressure, fTemperature);
354  if (energy < 0.01)
355  return 0.0;
356 
357  Double_t beta = TMath::Sqrt(1.0 - TMath::Power(M1 / (M1 + energy), 2));
358  Double_t ang[2] = {track->GetA(), track->GetB()};
359  Double_t norm = TMath::Sqrt(TMath::Tan(ang[0]) * TMath::Tan(ang[0]) + TMath::Tan(ang[1]) * TMath::Tan(ang[1]) + 1.0);
360  Double_t beta_array[3] = {beta * TMath::Tan(ang[0]) / norm, beta * TMath::Tan(ang[1]) / norm, beta * 1.0 / norm};
361 
362  TLorentzVector beam(0., 0., 0., M1);
363  beam.Boost(beta_array[0], beta_array[1], beta_array[2]);
364 
365  TLorentzVector target(0., 0., 0., M2);
366 
367  TLorentzVector cm_vec = beam + target;
368  TVector3 beta_cm = cm_vec.BoostVector();
369  beam.Boost(-beta_cm);
370  target.Boost(-beta_cm);
371 
372  return (beam.E() - beam.M()) + (target.E() - target.M());
373 }

454 Double_t TTGTIKProcessor::GetEcm_classic_kinematics(Double_t energy, Double_t theta) {
455  Double_t alpha = (M2 * (M1 + M2)) / (2.0 * M1);
456  Double_t beta = (M4 * (M3 + M4)) / (2.0 * M3);
457  Double_t qvalue = (M1 + M2) - (M3 + M4);
458 
459  Double_t v4 = TMath::Sqrt(2.0 * energy / M4);
460  Double_t b = (beta * v4 * TMath::Cos(theta)) / (alpha - beta);
461  Double_t c = (qvalue - beta * v4 * v4) / (alpha - beta);
462  Double_t D = b * b - c;
463  if (D < 0) {
464  if (TMath::Abs(D) < 1.0e-5) {
465  D = 0.0;
466  } else {
467  std::cerr << "b^2 - c = " << D << " < 0, det_energy : " << energy << ", theta : " << theta << std::endl;
468  return kInvalidD;
469  }
470  }
471 
472  Double_t vcm = -b + TMath::Sqrt(D);
473  if (vcm < 0) {
474  std::cerr << "vcm < 0! : vcm = " << -b << " + " << TMath::Sqrt(D) << ", det_energy : " << energy
475  << ", theta : " << theta << std::endl;
476  return kInvalidD;
477  }
478 
479  return alpha * vcm * vcm;
480 }

486 Double_t TTGTIKProcessor::GetCMAngle(Double_t ELab, Double_t Ecm, Double_t ALab) {
487  Double_t alpha = (M2 * (M1 + M2)) / (2.0 * M1);
488  Double_t beta = (M4 * (M3 + M4)) / (2.0 * M3);
489  Double_t qvalue = (M1 + M2) - (M3 + M4);
490 
491  Double_t v4 = TMath::Sqrt(2.0 * ELab / M4);
492  Double_t vcm = TMath::Sqrt(Ecm / alpha);
493  Double_t v4cm = TMath::Sqrt((Ecm + qvalue) / beta);
494 
495  Double_t theta_cm = TMath::ACos((v4 * TMath::Cos(ALab) - vcm) / v4cm);
496  return theta_cm;
497 }
