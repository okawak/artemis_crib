255 void TCRIBPIDProcessor::LoadPIDyaml() {
256  Info("Init", "Loading yaml file...");
257  FileStat_t info;
258  if (gSystem->GetPathInfo(fFileName.Data(), info) != 0) {
259  SetStateError(Form("File %s does not exist.", fFileName.Data()));
260  return;
261  }
262 
263  YAML::Node yaml_all = YAML::LoadFile(fFileName.Data());
264 
265 #if 0 // for debug
266  YAML::Emitter emitter;
267  emitter << yaml_all;
268  std::cout << emitter.c_str() << std::endl;
269  std::cout << yaml_all.size() << std::endl;
270 #endif
271 
272  // set ions
273  YAML::Node yaml_ions = yaml_all[kNodeKeyIons].as<YAML::Node>();
274  for (Size_t i = 0, n = yaml_ions.size(); i < n; i++) {
275  IntVec_t ion_config = yaml_ions[i].as<std::vector<int>>();
276  if (ion_config.size() != 4) {
277  std::cerr << "input yaml file: ion array size is not 4!" << std::endl;
278  return;
279  }
280 
281  ion_names.emplace_back(amdc::GetEl(ion_config[0]));
282  ion_massnum.emplace_back(ion_config[1]);
283  ion_charges.emplace_back(ion_config[2]);
284  ion_masses.emplace_back(amdc::Mass(ion_config[0], ion_config[1]) * amdc::amu);
285  ion_colors.emplace_back(ion_config[3]);
286  }
287 
288  // set BLD parameters
289  YAML::Node yaml_f1 = yaml_all[kNodeKeyF1Param].as<YAML::Node>();
290  f1_brho = yaml_f1[kNodeKeyBrho].as<double>();
291  degrader_mat = yaml_f1[kNodeKeyDegMat].as<std::string>();
292  degrader_thick = 1.0e-3 * yaml_f1[kNodeKeyDegThickness].as<double>();
293  rf_period = yaml_f1[kNodeKeyRfPeriod].as<double>();
294 
295  YAML::Node yaml_f2 = yaml_all[kNodeKeyF2Param].as<YAML::Node>();
296  f2_ppac_thickness = 1.0e-3 * yaml_f2[kNodeKeyPPACThickness].as<double>();
297  f2_ssd_thickness = 1.0e-3 * yaml_f2[kNodeKeySSDThickness].as<double>();
298  f2_rf_offset = yaml_f2[kNodeKeyRfOffset].as<double>();
299  f2_rf_range = yaml_f2[kNodeKeyRfRange].as<std::vector<double>>();
300  energy_range = yaml_f2[kNodeKeyERange].as<std::vector<double>>();
301 
302  YAML::Node yaml_f3 = yaml_all[kNodeKeyF3Param].as<YAML::Node>();
303  a_thickness = 1.0e-3 * yaml_f3[kNodeKeyAThickness].as<double>();
304  b_thickness = 1.0e-3 * yaml_f3[kNodeKeyBThickness].as<double>();
305  distance = yaml_f3[kNodeKeyDistance].as<double>();
306  trigger = yaml_f3[kNodeKeyTrigger].as<ushort>();
307  f3_rf_offset = yaml_f3[kNodeKeyRfOffset].as<double>();
308  tof_offset = yaml_f3[kNodeKeyTofOffset].as<double>();
309  f3_rf_range = yaml_f3[kNodeKeyRfRange].as<std::vector<double>>();
310  tof_range = yaml_f3[kNodeKeyTofRange].as<std::vector<double>>();
311 }
