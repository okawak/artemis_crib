122 void TF1PPACProcessor::Process() {
123  if (!fCategorizedData)
124  return;
125 
126  fPPACOut->Clear("C");
127  if (fDoSeparatePPACs) {
128  for (std::vector<TClonesArray *>::iterator it = fPPACArray.begin(); it != fPPACArray.end(); it++) {
129  (*it)->Clear("C");
130  }
131  }
132 
133  const TObjArray *const cat = (*fCategorizedData)->FindCategory(fCatID);
134  if (!cat)
135  return;
136  Int_t t[TPPACData::kNRAW], q[TPPACData::kNRAW];
137 
138  const Int_t n = cat->GetEntriesFast();
139  for (Int_t i = 0; i != n; ++i) {
140  const TObjArray *const det = static_cast<TObjArray *>(cat->At(i));
141  Bool_t idFlag = kFALSE;
142  Int_t detID = kInvalidI;
143 
144  for (Int_t j = 0; j != TPPACData::kNRAW; ++j) {
145 
146  t[j] = q[j] = kInvalidI;
147 
148  const TObjArray *const tArray = static_cast<TObjArray *>(det->At(j));
149  const TObjArray *const qArray = static_cast<TObjArray *>(det->At(j + TPPACData::kNRAW));
150 
151  if (tArray && tArray->GetEntriesFast()) {
152  TRawDataObject *const tHit = static_cast<TRawDataObject *>(tArray->At(0));
153  if (!tHit)
154  continue;
155  t[j] = tHit->GetValue();
156  if (!idFlag) {
157  detID = tHit->GetDetID();
158  idFlag = IsValid(detID);
159  }
160  }
161 
162  if (qArray && qArray->GetEntriesFast()) {
163  TRawDataObject *const qHit = static_cast<TRawDataObject *>(qArray->At(0));
164  if (qHit)
165  q[j] = qHit->GetValue();
166  }
167  }
168 
169  if (!idFlag)
170  continue;
171 
172  TPPACData *const ppac = static_cast<TPPACData *>(fPPACOut->ConstructedAt(fPPACOut->GetEntriesFast()));
173  ppac->SetDetID(detID);
174  for (Int_t j = 0; j != TPPACData::kNRAW; ++j) {
175  if (IsValid(t[j])) {
176  if (fTimingConverterArray) {
177  TConverterBase *conv = nullptr;
178  if (fHasEachChConverter) {
179  conv = fTimingConverterArray->at(detID * TPPACData::kNRAW + j);
180  } else {
181  conv = fTimingConverterArray->at(detID);
182  }
183  if (conv) {
184  ppac->SetT(conv->Convert(t[j]), j);
185  } else {
186  ppac->SetT(t[j], j);
187  }
188  } else {
189  ppac->SetT(t[j], j);
190  }
191  }
192 
193  if (IsValid(q[j])) {
194  if (fChargeConverterArray) {
195  TConverterBase *conv = nullptr;
196  if (fHasEachChConverter) {
197  conv = fChargeConverterArray->at(detID * TPPACData::kNRAW + j);
198  } else {
199  conv = fChargeConverterArray->at(detID);
200  }
201  ppac->SetQ(conv->Convert(q[j]), j);
202  } else {
203  ppac->SetQ(q[j], j);
204  }
205  }
206  }
207  if (fXConverterArray) {
208  ppac->SetXConverter(fXConverterArray->at(detID - DETID_ORIGIN));
209  }
210  if (fYConverterArray) {
211  ppac->SetYConverter(fYConverterArray->at(detID - DETID_ORIGIN));
212  }
213  ppac->Update();
214 
215  if (fDoSeparatePPACs && detID < (Int_t)fPPACArray.size()) {
216  Double_t X, Y;
217  TClonesArray *arr = fPPACArray[detID];
218  TPPACData *ppacdata = (TPPACData *)arr->ConstructedAt(arr->GetEntriesFast());
219  TPPACParameter *prm = (TPPACParameter *)fPPACParameter[detID];
220  ppac->Copy(*ppacdata);
221  Double_t dTX = (ppac->GetTX1() - ppac->GetTX2()) / (ppac->GetTX1() + ppac->GetTX2());
222  Double_t dTY = (ppac->GetTY1() - ppac->GetTY2()) / (ppac->GetTY1() + ppac->GetTY2());
223 
224  // if (delaylinecalibrated)
225  if (fDoInsideOffset) {
226  dTX += prm->GetInsideOffset(0);
227  dTY += prm->GetInsideOffset(1);
228  }
229 
230  // if (line calib ofset)
231  if (fDoOutsideOffset) {
232  dTX -= prm->GetOutsideOffset(0);
233  dTY -= prm->GetOutsideOffset(1);
234  }
235 
236  ppacdata->SetTXDiff(dTX);
237  ppacdata->SetTYDiff(dTY);
238  // exchange and calibration
239  X = dTX * prm->GetNs2mm(0) * 0.5;
240  Y = dTY * prm->GetNs2mm(1) * 0.5;
241  if (prm->GetExchangeXY()) {
242  Double_t tmp = X;
243  X = Y;
244  Y = tmp;
245  }
246  if (prm->GetReflection()) {
247  X *= -1;
248  }
249 
250  // geometry offset
251  if (fDoGeometryOffset) {
252  X -= prm->GetGeometryOffset(0);
253  Y -= prm->GetGeometryOffset(1);
254  }
255  ppacdata->ResetQualityBit(TDataObject::kInvalid);
256  if (!prm->IsInsideTXSum(ppacdata->GetTXSum())) {
257  ppacdata->SetQualityBit(TPPACData::kBadTXSum | TDataObject::kInvalid);
258  }
259  if (!prm->IsInsideTYSum(ppacdata->GetTYSum())) {
260  ppacdata->SetQualityBit(TPPACData::kBadTYSum | TDataObject::kInvalid);
261  }
262  ppacdata->SetX(X);
263  ppacdata->SetY(Y);
264  ppacdata->SetZ(prm->GetGeometryOffset(2));
265  ppacdata->Copy(*ppac);
266  }
267  }
