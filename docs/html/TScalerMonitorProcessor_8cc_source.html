166 void art::TScalerMonitorProcessor::ProcessShort(Long_t now) {
167  if (now - fCurrentShort > fShortInterval) {
168  const TScalerData *const scadata = static_cast<TScalerData *>(*fInData);
169  if (scadata->GetValue(fClock[0]) - fShortScatot[fClock[0]] <= 0) {
170  for (Int_t i = 0; i < SCALER_CH; i++) {
171  fShortScatot[i] = scadata->GetValue(i);
172  fShortScadiff[i] = scadata->GetValue(i);
173  }
174  fCurrentShort = now;
175  return;
176  }
177 
178  for (Int_t i = 0; i < SCALER_CH; i++) {
179  fShortScadiff[i] = scadata->GetValue(i) - fShortScatot[i];
180  fShortScatot[i] = scadata->GetValue(i);
181  }
182 
183  for (Int_t i = 0; i < SCALER_CH; i++) {
