134 void TNBodyReactionProcessor::Process() {
135  fOutData->Clear("C");
136  fOutReacData->Clear("C");
137 
138  if ((*fInData)->GetEntriesFast() != 1) {
139  SetStateError("input branch entry is not 1");
140  return;
141  }
142 
143  const TDataObject *const inData = static_cast<TDataObject *>((*fInData)->At(0));
144  const TParticleInfo *const Data = dynamic_cast<const TParticleInfo *>(inData);
145 
146  Double_t target_mass = amdc::Mass(fTargetAtmNum, fTargetMassNum) * amdc::amu; // MeV
147  TLorentzVector target_vec(0., 0., 0., target_mass);
148  TLorentzVector beam_vec = Data->GetLorentzVector();
149  Double_t beam_mass = amdc::Mass(fBeamNucleus[0], fBeamNucleus[1]) * amdc::amu; // MeV
150  Double_t beam_energy = Data->GetEnergy();
151 
152  // calculate reaction position
153  // target should be set at z=0 (entrance of gas target)
154  Double_t range = srim->Range(fBeamNucleus[0], fBeamNucleus[1], beam_energy,
155  std::string(fTargetName.Data()), fTargetPressure, 300.0);
156 
157  // determine using random number
158  Double_t reac_distance = GetRandomReactionDistance(range);
159  Double_t beam_energy_new = 0.0;
160  if (fTargetIsGas) {
161  beam_energy_new = srim->EnergyNew(fBeamNucleus[0], fBeamNucleus[1], beam_energy,
162  std::string(fTargetName.Data()), reac_distance, fTargetPressure, 300.0);
163  } else {
164  beam_energy_new = srim->EnergyNew(fBeamNucleus[0], fBeamNucleus[1], beam_energy,
165  std::string(fTargetName.Data()), reac_distance);
166  }
167 
168  Double_t reac_posz = 0.0;
169  if (fTargetIsGas) {
170  reac_posz = reac_distance * TMath::Cos(Data->GetTrack().GetA()) * TMath::Cos(Data->GetTrack().GetB());
171  }
172  Double_t reac_posx = Data->GetTrack().GetX(reac_posz);
173  Double_t reac_posy = Data->GetTrack().GetY(reac_posz);
174 
175  beam_vec = GetLossEnergyVector(beam_vec, beam_energy - beam_energy_new);
176 
177  // calculate tof in the target (mean value)
178  Double_t duration_beam = 0.0;
179  if (fTargetIsGas) {
180  duration_beam += reac_distance / (TMath::Sqrt(1.0 - TMath::Power(beam_mass / beam_vec.E(), 2.0)) * c);
181  } // if solid target, tof is almost zero
182 
183  DoubleVec_t reac_masses;
184  for (Int_t iPart = 0; iPart < fDecayNum; iPart++) {
185  Double_t mass = amdc::Mass(fReacAtmNum[iPart], fReacMassNum[iPart]) * amdc::amu; // MeV
186  mass += fExciteLevel[iPart];
187  reac_masses.emplace_back(mass);
188  }
189 
190  TLorentzVector compound_vec = beam_vec + target_vec;
191 
192  // to CM system (only beam_vec and target_vec)
193  TVector3 beta_vec = compound_vec.BoostVector();
194  beam_vec.Boost(-beta_vec);
195  target_vec.Boost(-beta_vec);
196  Double_t energy_cm = (beam_vec.E() - beam_vec.M()) + (target_vec.E() - target_mass);
197 
198  // need to change MeV to GeV
199  compound_vec *= 0.001;
200  DoubleVec_t GeVMass;
201  for (decltype(reac_masses.size()) i = 0; i < reac_masses.size(); i++) {
202  GeVMass.emplace_back(reac_masses[i] * 0.001);
203  }
204 
205  Bool_t isOkay = event.SetDecay(compound_vec, fDecayNum, GeVMass.data());
206  if (!isOkay) {
207  std::cerr << "forbidden kinematics" << std::endl;
208  }
209  event.Generate();
210 
211  Double_t theta_cm = 0.0;
212  for (Int_t iPart = 0; iPart < fDecayNum; ++iPart) {
213  TLorentzVector reac_vec = *event.GetDecay(iPart);
214  // need to change GeV to MeV
215  reac_vec *= 1000.;
216 
217  TParticleInfo *outData = static_cast<TParticleInfo *>(fOutData->ConstructedAt(iPart));
218  outData->SetID(iPart);
219  outData->SetMassNumber(fReacMassNum[iPart]);
220  outData->SetAtomicNumber(fReacAtmNum[iPart]);
221  outData->SetCharge(fReacAtmNum[iPart]); // no need?
222  outData->SetTrack(reac_posx, reac_posy, reac_posz, TMath::ATan(reac_vec.Px() / reac_vec.Pz()),
223  TMath::ATan(reac_vec.Py() / reac_vec.Pz()));
224  if (fTargetIsGas) {
225  outData->SetLorentzVector(reac_vec);
226  outData->SetEnergy(reac_vec.E() - reac_masses[iPart]);
227  outData->SetCurrentZ(reac_posz);
228  outData->SetZeroTime(); // initialize
229  outData->AddTime(duration_beam);
230  } else {
231  // for the solid target, ignore the angle
232  Double_t first_energy = reac_vec.E() - reac_masses[iPart];
233  if (first_energy > 0.01) {
234  Double_t out_thickness = fTargetThickness - reac_distance;
235  if (out_thickness < 0.0) {
236  out_thickness = 0.0;
237  }
238  TLorentzVector out_vec =
239  GetLossEnergyVector(reac_vec,
240  first_energy - srim->EnergyNew(fReacAtmNum[iPart], fReacMassNum[iPart], first_energy, std::string(fTargetName.Data()), out_thickness));
241 
242  outData->SetLorentzVector(out_vec);
243  outData->SetEnergy(out_vec.E() - reac_masses[iPart]);
244  outData->SetCurrentZ(fTargetThickness);
245  outData->SetZeroTime();
246  } else {
247  // outData->SetLorentzVector(reac_vec);
248  outData->SetLorentzVector(0.0, 0.0, 0.0, reac_masses[iPart]);
249  outData->SetEnergy(0.0);
250  outData->SetCurrentZ(reac_posz);
251  outData->SetZeroTime();
252  }
253  }
254 
255  // compound_vec and reac_vec is LAB system, convert to CM system
256  reac_vec.Boost(-beta_vec);
257  outData->SetThetaCM(reac_vec.Theta() / deg2rad);
258  outData->SetPhiCM(reac_vec.Phi() / deg2rad);
259  if (iPart == 0) {
260  theta_cm = (Data->GetLorentzVector()).Angle(reac_vec.Vect()) / deg2rad;
261  }
262  }
263 
264  TReactionInfo *outReacData = static_cast<TReactionInfo *>(fOutReacData->ConstructedAt(0));
265  outReacData->SetID(0);
266  outReacData->SetExEnergy(fExciteLevel[0]);
267  outReacData->SetEnergy(energy_cm);
268  outReacData->SetTheta(theta_cm);
269  outReacData->SetXYZ(reac_posx, reac_posy, reac_posz);

272 void TNBodyReactionProcessor::InitGeneratingFunc() {
273  gr_generating_func = new TGraph();
274  gr_generating_func_inv = new TGraph();
275 
276  // simplize range
277  auto get_range = [&](Double_t e) {
278  return srim->Range(fBeamNucleus[0], fBeamNucleus[1], e, std::string(fTargetName.Data()), fTargetPressure, 300.0);
279  };
280 
281  // get dE/dx (x : range)
282  auto dedx = [&](Double_t e) {
283  const Double_t h = 0.1;
284  Double_t dxde = (get_range(e + h) - get_range(e)) / h;
285  return 1.0 / dxde;
286  };
287 
288  auto gr_density_func = new TGraph();
289  Int_t index = 0; // index for TGraph
290  Double_t xmin = 0.0, xmax = 0.0;
291  std::ifstream fin(fCSDataPath.Data());
292  if (!fin) {
293  Info("Init", "no input cross section file, use uniform energy distribution");
294  for (auto e = 0.0; e < fBeamEnergy * 1.5; e += 0.5) {
295  Double_t range = get_range(e);
296  gr_density_func->SetPoint(index, range, dedx(e));
297  index++;
298  xmax = range;
299  if (index == 1) {
300  xmin = range;
301  }
302  }
303  } else {
304  Info("Init", "get cross section file: %s", fCSDataPath.Data());
305  Double_t ene_factor = 0.0;
306  if (fCSType == 0) {
307  ene_factor = amdc::Mass(fBeamNucleus[0], fBeamNucleus[1]) / amdc::Mass(fTargetAtmNum, fTargetMassNum);
308  } else if (fCSType == 1) {
309  ene_factor = 1.0;
310  } else if (fCSType == 2) {
311  ene_factor = (amdc::Mass(fBeamNucleus[0], fBeamNucleus[1]) + amdc::Mass(fTargetAtmNum, fTargetMassNum)) /
312  amdc::Mass(fTargetAtmNum, fTargetMassNum);
313  } else {
314  SetStateError("CrossSectionType should be 0, 1 or 2");
315  return;
316  }
317 
318  std::string line;
319  std::regex commentRegex("#.*");
320  while (std::getline(fin, line)) {
321  if (line.empty() || line[0] == '#') {
322  continue;
323  }
324  line = std::regex_replace(line, commentRegex, "");
325 
326  std::vector<TString> values;
327  TString line_copy = line;
328  while (line_copy.Contains(' ')) {
329  Int_t index = line_copy.First(' ');
330  TString line_val, line_remain;
331  line_val = line_copy(0, index);
332  line_remain = line_copy(index + 1, line_copy.Length());
333  line_copy = line_remain;
334  if (line_val == "") {
335  continue;
336  }
337  values.emplace_back(line_val);
338  }
339  Double_t e = ene_factor * values[0].Atof();
340  Double_t cs = values[1].Atof() / ene_factor;
341  Double_t range = get_range(e);
342  gr_density_func->SetPoint(index, range, cs * dedx(e));
343  index++;
344  xmax = range;
345  if (index == 1) {
346  xmin = range;
347  }
348  }
349  }
350 
351  auto f = new TF1("f", [&](double *x, double *p) { return p[0] * gr_density_func->Eval(x[0], nullptr, "S"); }, xmin, xmax, 1);
352  f->SetParameter(0, 1.0);
353 
354  Info("Init", "suppressing the error message in this initialization process");
355  Info("Init", "please wait for a moment...");
356 
357  // suppress error message when calculating the integral value
358  freopen("/dev/null", "w", stderr);
359  index = 0;
360  for (auto x = xmin; x < xmax; x += 1.0) {
361  gr_generating_func->SetPoint(index, x, f->Integral(xmin, x));
362  gr_generating_func_inv->SetPoint(index, f->Integral(xmin, x), x);
363  index++;
364  }
365  freopen("/dev/tty", "w", stderr);
366 
367  Info("Init", "total cross section (arbitrary unit or mb)");
368  Info("Init", "\t%lf", gr_generating_func->Eval(get_range(fBeamEnergy), nullptr, "S"));
369 }

371 Double_t TNBodyReactionProcessor::GetRandomReactionDistance(Double_t range) {
372  Double_t random_x = 0.0;
373  Double_t max_x = gr_generating_func->Eval(range, nullptr, "S");
374  if (range < fTargetThickness) {
375  random_x = max_x * gRandom->Uniform();
376  } else {
377  Double_t limit_range = range - fTargetThickness;
378  Double_t min_x = gr_generating_func->Eval(limit_range, nullptr, "S");
379  random_x = gRandom->Uniform(min_x, max_x);
380  }
381  Double_t distance = gr_generating_func_inv->Eval(random_x, nullptr, "S");
382 
383  return range - distance;
384 }

386 TLorentzVector TNBodyReactionProcessor::GetLossEnergyVector(TLorentzVector vec, Double_t eloss) {
387  Double_t factor =
388  ((vec.E() - eloss) * (vec.E() - eloss) - vec.M() * vec.M()) / (vec.E() * vec.E() - vec.M() * vec.M());
389  factor = TMath::Sqrt(factor);
390  TLorentzVector new_vec(factor * vec.Px(), factor * vec.Py(), factor * vec.Pz(), vec.E() - eloss);
391  return new_vec;
392 }

396 // lorentz invariant

402 // beta = v/c
