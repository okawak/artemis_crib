148 void TDetectParticleProcessor::Process() {
149  fOutData->Clear("C");
150  TGeoManager *geom = static_cast<TGeoManager *>(*fInGeom);
151 
152  for (Int_t iData = 0; iData < (*fInData)->GetEntriesFast(); ++iData) {
153  const TDataObject *const inData = static_cast<TDataObject *>((*fInData)->At(iData));
154  const TParticleInfo *const Data = dynamic_cast<const TParticleInfo *>(inData);
155 
156  TTelescopeData *outData = static_cast<TTelescopeData *>(fOutData->ConstructedAt(iData));
157  outData->SetID(iData);
158 
159  Double_t energy = Data->GetEnergy();
160  if (energy < 0.01) {
161  continue; // stop in the target
162  }
163 
164  Double_t current_z = Data->GetCurrentZ();
165  TTrack track = Data->GetTrack();
166  TVector3 first_position(track.GetX(current_z), track.GetY(current_z), current_z);
167  TVector3 velocity = (Data->GetLorentzVector()).Vect().Unit();
168 
169  // check if beam particle hits the detector
170  geom->SetCurrentPoint(first_position.X(), first_position.Y(), first_position.Z());
171  geom->SetCurrentDirection(velocity.X(), velocity.Y(), velocity.Z());
172 
173  // exclude TOP boundary
174  geom->GetCurrentNode();
175  geom->FindNode();
176 
177  // geom->FindNextBoundary(10000.0); // need to set some values??
178  geom->FindNextBoundary();
179  Double_t distance = geom->GetStep();
180  geom->Step();
181  Bool_t isHit = geom->IsStepEntering();
182  if (!isHit) {
183  continue;
184  }
185 
186  TString hitname = geom->GetPath(); // ex. hitname = /TOP_1/tel1_0, /TOP_1/tel4_3
187  auto index_hitpath = hitname.Last('_');
188  TString det_id = hitname(index_hitpath + 1, hitname.Length()); // get last number /TOP_1/tel1_0 -> 0
189  if (hitname.Length() < 8 || det_id.Atoi() >= (*fDetectorPrm)->GetEntriesFast()) {
190  // /TOP_1/ -> length = 7
191  continue;
192  }
193 
194  if (fTargetIsGas) {
195  energy = srim->EnergyNew(Data->GetAtomicNumber(), Data->GetMassNumber(), energy,
196  std::string(fTargetName.Data()), distance, fTargetPressure, 300.0);
197  if (energy < 0.01) {
198  continue; // stop in the target
199  }
200  }
201 
202  outData->SetTelID(det_id.Atoi() + 1);
203  TVector3 det_position(distance * velocity.X(), distance * velocity.Y(), distance * velocity.Z());
204  det_position += first_position;
205  outData->SetPosition(det_position);
206 
207  TParameterObject *inPrm = static_cast<TParameterObject *>((*fDetectorPrm)->At(det_id.Atoi()));
208  TDetectorParameter *Prm = dynamic_cast<TDetectorParameter *>(inPrm);
209  outData->SetN(Prm->GetN());
210 
211  det_position -= TVector3(Prm->GetCenterRotPos(0), Prm->GetCenterRotPos(1), Prm->GetCenterRotPos(2));
212  det_position.RotateY(-Prm->GetAngle());
213 
214  outData->SetXID(GetStripID(det_position.X(), Prm->GetStripNum(0), Prm->GetSize(0)));
215  outData->SetYID(GetStripID(det_position.Y(), Prm->GetStripNum(1), Prm->GetSize(1)));
216  if (outData->GetXID() == -1 || outData->GetYID() == -1) {
217  continue;
218  }
219 
220  // calculate TOF
221  Double_t ion_mass = amdc::Mass(Data->GetAtomicNumber(), Data->GetMassNumber()) * amdc::amu; // MeV
222  Double_t duration = distance / (TMath::Sqrt(1.0 - TMath::Power(ion_mass / (ion_mass + energy), 2.0)) * c); // ns
223  duration += Data->GetDurationTime();
224  outData->PushTimingArray(duration);
225 
226  // caliculate energy
227  Double_t energy_total = 0.0;
228  for (auto iMat = 0; iMat < Prm->GetN(); iMat++) {
229  if (energy > 0.01) {
230  Double_t new_energy = srim->EnergyNew(Data->GetAtomicNumber(), Data->GetMassNumber(), energy,
231  std::string((Prm->GetMaterial(iMat)).Data()), Prm->GetThickness(iMat));
232  energy_total += energy - new_energy;
233  outData->PushEnergyArray(energy - new_energy);
234  energy = new_energy;
235  } else {
236  outData->PushEnergyArray(0.0);
237  }
238  }
239  outData->SetEtotal(gRandom->Gaus(energy_total, energy_total * fResolution[det_id.Atoi()]));
240 
241  // caliculate LAB angle
242  const TDataObject *const inTrackData = static_cast<TDataObject *>((*fInTrackData)->At(0));
243  const TTrack *const TrackData = dynamic_cast<const TTrack *>(inTrackData);
244  TVector3 beam(TrackData->GetX(1.0) - TrackData->GetX(0.0), TrackData->GetY(1.0) - TrackData->GetY(0.0), 1.0);
245  Double_t theta = beam.Angle(velocity);
246  outData->SetTheta_L(theta * TMath::RadToDeg());
247  }
