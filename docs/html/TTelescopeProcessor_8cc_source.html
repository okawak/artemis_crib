144 void TTelescopeProcessor::Process() {
145  fOutData->Clear("C");
146 
147  Int_t nData1 = (*fInData1)->GetEntriesFast();
148  Int_t nData2 = (*fInData2)->GetEntriesFast();
149  Int_t nData3 = (*fInData3)->GetEntriesFast();
150 
151  // if no hit, do nothing
152  if (nData1 == 0 && nData2 == 0 && nData3 == 0) {
153  return;
154  }
155 
156  TTelescopeData *outData = static_cast<TTelescopeData *>(fOutData->ConstructedAt(0));
157  outData->Clear();
158  outData->SetID(0); // always 0
159  outData->SetTelID(fTelID);
160  if (fHasDetPrm && fHasTargetPrm) {
161  outData->SetN(fDetParameter->GetN());
162  } else {
163  outData->SetN(DEFAULT_SSD_MAX_NUMBER);
164  }
165 
166  Double_t Etotal = 0.0;
167  Double_t dE = 0.0;
168 
169  // dEX process
170  // search the highest channel
171  Int_t dEXID = -1;
172  Double_t dEX_tmp = 0.0;
173  for (Int_t iData1 = 0; iData1 < nData1; ++iData1) {
174  const TDataObject *const inData1 = static_cast<TDataObject *>((*fInData1)->At(iData1));
175  const TTimingChargeData *const Data1 = dynamic_cast<const TTimingChargeData *>(inData1);
177  if (dEX_tmp < Data1->GetCharge() && Data1->GetCharge() < 50.0) { // overflow signal should have very large value
178  dEXID = iData1;
179  dEX_tmp = Data1->GetCharge();
180  }
181  }
182 
183  // fill the highest channel
184  if (nData1 != 0 && dEXID != -1) {
185  const TDataObject *const inData1 = static_cast<TDataObject *>((*fInData1)->At(dEXID));
186  const TTimingChargeData *const Data1 = dynamic_cast<const TTimingChargeData *>(inData1);
187  Double_t energy = Data1->GetCharge();
188  Double_t timing = Data1->GetTiming();
189 
190  // judge if the event is pedestal or not
191  if (fHasDetPrm) {
192  if (energy < fDetParameter->GetPedestal(0)) { // first layer
193  energy = 0.0;
194  }
195  }
196 
197  outData->SetXID(Data1->GetDetID());
198  if (fIsDSSSD) {
199  outData->SetdE(energy); // now fdE is used the value of dEX
200  } else {
201  dE += energy;
202  }
203  outData->SetdEX(energy);
204  outData->SetTelXTiming(timing);
205  outData->PushEnergyArray(energy);
206  outData->PushTimingArray(timing);
207  Etotal += energy;
208  } else {
209  outData->PushEnergyArray(0.0);
210  outData->PushTimingArray(kInvalidD);
211  }
212 
213  // dEY process
214  // search the highest channel
215  Int_t dEYID = -1;
216  Double_t dEY_tmp = 0.0;
217  for (Int_t iData2 = 0; iData2 < nData2; ++iData2) {
218  const TDataObject *const inData2 = static_cast<TDataObject *>((*fInData2)->At(iData2));
219  const TTimingChargeData *const Data2 = dynamic_cast<const TTimingChargeData *>(inData2);
221  if (dEY_tmp < Data2->GetCharge() && Data2->GetCharge() < 50.0) { // overflow signal should have very large value
222  dEYID = iData2;
223  dEY_tmp = Data2->GetCharge();
224  }
225  }
226 
227  // fill the highest channel
228  if (nData2 != 0 && dEYID != -1) {
229  const TDataObject *const inData2 = static_cast<TDataObject *>((*fInData2)->At(dEYID));
230  const TTimingChargeData *const Data2 = dynamic_cast<const TTimingChargeData *>(inData2);
231  Double_t energy = Data2->GetCharge();
232  Double_t timing = Data2->GetTiming();
233 
234  // judge if this event is pedestal or not
235  if (fHasDetPrm) {
236  if (fIsDSSSD) {
237  if (energy < fDetParameter->GetPedestal(0)) { // first layer
238  energy = 0.0;
239  }
240  } else {
241  if (energy < fDetParameter->GetPedestal(1)) { // second layer
242  energy = 0.0;
243  }
244  }
245  }
246 
247  outData->SetYID(Data2->GetDetID());
248  if (!fIsDSSSD) {
249  dE += energy;
250  outData->SetdE(dE);
251  outData->PushEnergyArray(energy);
252  outData->PushTimingArray(timing);
253  Etotal += energy;
254  }
255  outData->SetdEY(energy);
256  outData->SetTelYTiming(timing);
257  } else if (!fIsDSSSD) {
258  outData->PushEnergyArray(0.0);
259  outData->PushTimingArray(kInvalidD);
260  }
261 
262  // geometry process
264  if (IsValid(outData->GetXID()) && IsValid(outData->GetYID()) && fHasDetPrm && fHasTargetPrm) {
265  Double_t target_z = fTargetParameter->GetZ();
266  TVector3 center_rot(fDetParameter->GetCenterRotPos(0), fDetParameter->GetCenterRotPos(1), fDetParameter->GetCenterRotPos(2));
267  TVector3 offset(fDetParameter->GetOffset(0), fDetParameter->GetOffset(1), fDetParameter->GetOffset(2));
268  DoubleVec_t size = fDetParameter->GetSize();
269  IntVec_t stripN = fDetParameter->GetStripNum();
270  Int_t xid = outData->GetXID();
271  Int_t yid = outData->GetYID();
272 
273  // (x, y) position before rotation
274  Double_t x = size[0] * (2.0 * (Double_t)xid + 1.0 - (Double_t)stripN[0]) / (2.0 * (Double_t)stripN[0]);
275  if (fUseRandom) {
276  x += gRandom->Uniform(-size[0] / (2.0 * (Double_t)stripN[0]), size[0] / (2.0 * (Double_t)stripN[0]));
277  }
278  Double_t y = size[1] * (2.0 * (Double_t)yid + 1.0 - (Double_t)stripN[1]) / (2.0 * (Double_t)stripN[1]);
279  if (fUseRandom) {
280  y += gRandom->Uniform(-size[1] / (2.0 * (Double_t)stripN[1]), size[1] / (2.0 * (Double_t)stripN[1]));
281  }
282 
283  // rotate the detector
284  TVector3 det_pos(x, y, fDetParameter->GetDistance());
285  det_pos += offset;
286  det_pos.RotateY(fDetParameter->GetAngle()); // radian
287  det_pos += center_rot;
288  outData->SetPosition(det_pos);
289 
290  TVector3 target(0., 0., target_z);
291  TVector3 beam(0., 0., 1.); // assume beam center!
292  Double_t theta = beam.Angle(det_pos - target);
293  outData->SetTheta_L(theta * TMath::RadToDeg());
294  }
295 
296  // Thick SSDs process
297  Double_t E = 0.0;
298  Int_t NE;
299  if (fHasDetPrm && fHasTargetPrm) {
300  NE = fIsDSSSD ? fDetParameter->GetN() - 1 : fDetParameter->GetN() - 2;
301  } else {
302  NE = fIsDSSSD ? DEFAULT_SSD_MAX_NUMBER - 1 : DEFAULT_SSD_MAX_NUMBER - 2;
303  }
304  Int_t index = fIsDSSSD ? 1 : 2;
305 
306  if (NE < nData3) {
307  Warning("process", "Thick SSD number is wrong between actual data and define in get/expname.yaml");
308  }
309 
310  for (Int_t iE = 0; iE < NE; iE++) {
311  Int_t itr = -1;
312  // get "iE" index object
314  for (Int_t iData3 = 0; iData3 < nData3; ++iData3) {
315  const TDataObject *const inData3 = static_cast<TDataObject *>((*fInData3)->At(iData3));
316  const TTimingChargeData *const Data3 = dynamic_cast<const TTimingChargeData *>(inData3);
317  if (Data3->GetDetID() == iE) {
318  itr = iData3;
319  break;
320  }
321  }
322 
323  if (itr < 0) {
324  outData->PushEnergyArray(0.0);
325  outData->PushTimingArray(kInvalidD);
326  } else {
327  const TDataObject *const inData3 = static_cast<TDataObject *>((*fInData3)->At(itr));
328  const TTimingChargeData *const Data3 = dynamic_cast<const TTimingChargeData *>(inData3);
329  Double_t energy = Data3->GetCharge();
330  Double_t timing = Data3->GetTiming();
331 
332  // judge if the event is pedestal or not
333  if (fHasDetPrm) {
334  if (energy < fDetParameter->GetPedestal(index + iE)) {
335  energy = 0.0;
336  }
337  }
338 
339  outData->PushEnergyArray(energy);
340  outData->PushTimingArray(timing);
341  E += energy;
342  Etotal += energy;
343  }
344  }
345 
346  outData->SetE(E);
347  outData->SetEtotal(Etotal);
348 }
