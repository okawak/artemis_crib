128  // TGeo material setting
129  YAML::Node yaml_mat = yaml_all[kNodeKeyMaterial].as<YAML::Node>();
130  std::vector<TGeoMaterial *> mat_vec;
131  std::vector<TGeoMedium *> med_vec;
132  for (decltype(yaml_mat.size()) i = 0; i < yaml_mat.size(); i++) {
133  TString name = yaml_mat[i][kNodeKeyName].as<std::string>();
134  Double_t atm_mass = yaml_mat[i][kNodeKeyAMass].as<double>();
135  Double_t atm_num = yaml_mat[i][kNodeKeyANum].as<double>();
136  Double_t density = yaml_mat[i][kNodeKeyDensity].as<double>();
137  mat_vec.emplace_back(new TGeoMaterial(name.Data(), atm_mass, atm_num, density));
138  med_vec.emplace_back(new TGeoMedium(name.Data(), i + 1, mat_vec[mat_vec.size() - 1]));
139  }
140 
141  // TGeo top setting
142  YAML::Node yaml_top = yaml_all[kNodeKeyVolume][kNodeKeyTop].as<YAML::Node>();
143  TString top_name = yaml_top[kNodeKeyName].as<std::string>();
144  Int_t top_mat_id = yaml_top[kNodeKeyMaterial].as<int>();
145  DoubleVec_t top_size = yaml_top[kNodeKeySize].as<std::vector<double>>();
146  if (top_size.size() != 3) {
147  SetStateError("input yaml error");
148  return;
149  }
150 
151  TGeoVolume *top =
152  fGeom->MakeBox(top_name.Data(), med_vec[top_mat_id], top_size[0] / 2., top_size[1] / 2., top_size[2] / 2.);
153  fGeom->SetTopVolume(top);
154 
155  // TGeo detector setting
156  YAML::Node yaml_prm = yaml_all[kNodeKeyConposition][kNodeKeyDetector].as<YAML::Node>();
157  YAML::Node yaml_det = yaml_all[kNodeKeyVolume][kNodeKeyDetector].as<YAML::Node>();
158  if (yaml_prm.size() != yaml_det.size()) {
159  SetStateError("in yaml, conposition number and detector number is different");
160  return;
161  }
162 
163  for (decltype(yaml_det.size()) i = 0; i < yaml_det.size(); i++) {
164  if (yaml_prm[i][kNodeKeyName].as<std::string>() != yaml_det[i][kNodeKeyName].as<std::string>()) {
165  SetStateError("in yaml, set same order for composition and detector");
166  return;
167  }
168  TString det_name = yaml_det[i][kNodeKeyName].as<std::string>();
169  Int_t det_mat_id = yaml_det[i][kNodeKeyMaterial].as<int>();
170  DoubleVec_t det_size = yaml_det[i][kNodeKeySize].as<std::vector<double>>();
171  if (det_size.size() != 3) {
172  SetStateError("input yaml error, detector volume size must be 3D");
173  return;
174  }
175 
176  // currently only support box type detector
177  if (yaml_det[i][kNodeKeyType].as<std::string>() != "box") {
178  SetStateError("Currently only support only box type");
179  return;
180  }
181 
182  TGeoVolume *det =
183  fGeom->MakeBox(det_name.Data(), med_vec[det_mat_id], det_size[0] / 2., det_size[1] / 2., det_size[2] / 2.);
184  DoubleVec_t rot_point = yaml_prm[i][kNodeKeyCenterRot].as<std::vector<double>>();
185  DoubleVec_t offset = yaml_prm[i][kNodeKeyOffset].as<std::vector<double>>();
186  IntVec_t det_strip = yaml_prm[i][kNodeKeyStrip].as<std::vector<int>>();
187  if (rot_point.size() != 3 || offset.size() != 3 || det_strip.size() != 2) {
188  SetStateError("input yaml error, detector conposition setting is wrong");
189  return;
190  }
191  Double_t distance = yaml_prm[i][kNodeKeyDistance].as<double>();
192  Double_t angle = yaml_prm[i][kNodeKeyAngle].as<double>() * deg2rad;
193 
194  TVector3 det_pos(offset[0], offset[1], distance + offset[2]);
195  det_pos.RotateY(angle);
196  TVector3 rot_point_vec(rot_point[0], rot_point[1], rot_point[2]);
197  det_pos += rot_point_vec;
198 
199  TGeoCombiTrans *det_trans =
200  new TGeoCombiTrans(det_pos.X(), det_pos.Y(), det_pos.Z(), new TGeoRotation("rot", 90.0, angle / deg2rad, 0.0));
201  top->AddNode(det, i, det_trans);
202 
203  // parameter input
204  TDetectorParameter *prm = static_cast<TDetectorParameter *>(fDetParameterArray->ConstructedAt(i));
205  DoubleVec_t thickness = yaml_prm[i][kNodeKeyThickness].as<std::vector<double>>();
206  DoubleVec_t pedestal = yaml_prm[i][kNodeKeyPedestal].as<std::vector<double>>();
207  if (thickness.size() != pedestal.size()) {

209  return;
210  }

212  StringVec_t material_vec;
213  for (decltype(thickness.size()) j = 0; j < thickness.size(); j++) {
214  if (material.size() == 1) {
215  material_vec.emplace_back(material[0]);
216  } else if (material.size() == thickness.size()) {
217  material_vec.emplace_back(material[j]);
218  } else {

220  return;
221  }
222  }
223
