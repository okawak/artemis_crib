45 void TEffectiveThickness::Init(TEventCollection *col)
46 {
47  if (fParticleNames.size() != 2 || fParticleMass.size() != 2) {
48  SetStateError("input steering yaml: not 2 particles (beam and target)");
49  return;
50  }
51 
52  Info("Init", "reaction: %s + %s", fParticleNames[0].Data(), fParticleNames[1].Data());
53  for (Int_t i = 0; i < fParticleMass.size(); i++) {
54  Info("Init", "\t%s: %lf amu", fParticleNames[i].Data(), fParticleMass[i]);
55  fParticleMass[i] *= TArtAtomicMassTable::kAtomicMassUnit; // amu to MeV
56  }
57 
58  Info("Init", "target: %s (%lf g/cm3)", fTarget.Data(), fDensity);
59 
60  fElossTable = new SRIMData();
61  fElossBeam = fElossTable->GetTable(fParticleNames[0], fTarget);
62 
63  TGraph *g_mm = new TGraph();
64  TGraph *g_g = new TGraph();
65  TGraph *g_atm = new TGraph();
66  Int_t itr = 0;
67  for (Double_t energy = fRange[0]; energy < fRange[1]; energy += 0.01) {
68  Double_t e_min = energy - fDeltaE / 2.0;
69  Double_t e_max = energy + fDeltaE / 2.0;
70  if (e_min < 0.0) {
71  e_min = 0.0;
72  }
73  // thickness (mm)
74  Double_t thickness = fElossBeam->GetRange(Ecm2Elab(e_max)) - fElossBeam->GetRange(Ecm2Elab(e_min));
75  g_mm->SetPoint(itr, energy, thickness);
76 
77  // thickness ((mm->cm) * g/cm3) -> g/cm2
78  g_g->SetPoint(itr, energy, thickness * 1.0e-1 * fDensity);
79 
80  // thickness ((mm->cm) * g/cm3 * /mol / g/mol) -> /cm2
81  g_atm->SetPoint(itr, energy, thickness * 1.0e-1 * fDensity * kAvogadro / fTargetA);
82 
83  itr++;
84  }
85  g_mm->SetTitle(Form("thickness(mm, DeltaE=%.2lf MeV);Ecm (MeV);Effective thickness (mm)", fDeltaE));
86  g_g->SetTitle(Form("thickness(g_cm2, DeltaE=%.2lf MeV);Ecm (MeV);Effective thickness (g/cm2)", fDeltaE));
87  g_atm->SetTitle(Form("thickness(atom_cm2, DeltaE=%.2lf MeV);Ecm (MeV);Effective thickness (atom/cm2)", fDeltaE));
88 
89  gDirectory->Add(g_mm);
90  gDirectory->Add(g_g);
91  gDirectory->Add(g_atm);
92 
93  if (fOutputName != "") {
94  Util::PrepareDirectoryFor(fOutputName);
95  TFile *file = new TFile(fOutputName, "recreate");
96  if (!file) {
97  std::cerr << "ERROR : cannot open " << fOutputName << std::endl;
98  return;
99  }
100  g_mm->Write("thickness_mm");
101  g_g->Write("thickness_g_cm2");
102  g_atm->Write("thickness_atom_cm2");
103 
104  file->Close();
105 
106  Info("Init", "make histogram rootfile: %s", fOutputName.Data());
107  }
108 }

112 Double_t TEffectiveThickness::Ecm2Elab(Double_t energy)
113 {
114  // kinematics
115  Double_t tmp =
116  energy * energy + 2.0 * (energy * (fParticleMass[0] + fParticleMass[1]) + fParticleMass[0] * fParticleMass[1]);
117  return tmp / (2.0 * fParticleMass[1]) - fParticleMass[0];
118 }
119 
120 // ============================================================
121 // kinematics
122 // using Lorentz invariant (c = 1 unit)
123 // Ecm^2 = (E1lab + E2lab)^2 - (p1lab + p2lab)^2
124 // (Ekincm + m1 + m2)^2 = (E1lab + m2)^2 - p1lab^2
125 // (Ekincm + m1 + m2)^2 = (E1lab + m2)^2 - (E1lab^2 - m1^2)
126 // ...
127 // E1kinlab = E1lab - m1 = ...
128 // from this relationship, the Ecm2Elab is calculated
